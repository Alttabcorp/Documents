\section{Cronograma}

\subsection{Metodologia de Desenvolvimento}

O projeto seguira a metodologia Agile com sprints de 2 semanas, utilizando praticas de desenvolvimento iterativo e incremental.

\subsubsection{Fases do Projeto}
\begin{itemize}
    \item \textbf{Fase 1}: Planejamento e Design (4 semanas)
    \item \textbf{Fase 2}: Desenvolvimento MVP (12 semanas)
    \item \textbf{Fase 3}: Testes e Refinamentos (4 semanas)
    \item \textbf{Fase 4}: Deploy e Lancamento (2 semanas)
    \item \textbf{Fase 5}: Pos-lancamento e Melhorias (ongoing)
\end{itemize}

\subsection{Cronograma Detalhado}

\subsubsection{Fase 1: Planejamento e Design (Semanas 1-4)}

\paragraph{Semana 1-2: Analise e Especificacao}
\begin{itemize}
    \item Levantamento detalhado de requisitos
    \item Definicao da arquitetura do sistema
    \item Criacao de user stories
    \item Planejamento tecnico detalhado
    \item Setup do ambiente de desenvolvimento
\end{itemize}

\paragraph{Semana 3-4: Design e Prototipagem}
\begin{itemize}
    \item Criacao do design system
    \item Wireframes de todas as telas
    \item Prototipos interativos no Figma
    \item Validacao de UX com usuarios
    \item Definicao da identidade visual
\end{itemize}

\subsubsection{Fase 2: Desenvolvimento MVP (Semanas 5-16)}

\paragraph{Sprint 1 (Semanas 5-6): Infraestrutura}
\begin{itemize}
    \item Setup do projeto backend (Node.js + TypeScript)
    \item Configuracao do banco de dados PostgreSQL
    \item Setup do projeto frontend React
    \item Configuracao de CI/CD basico
    \item Implementacao de autenticacao JWT
\end{itemize}

\paragraph{Sprint 2 (Semanas 7-8): Gestao de Usuarios}
\begin{itemize}
    \item Sistema de registro e login
    \item Recuperacao de senha
    \item Perfis de usuario
    \item Gestao de casais/contas compartilhadas
    \item Testes unitarios basicos
\end{itemize}

\paragraph{Sprint 3 (Semanas 9-10): Modulo de Receitas}
\begin{itemize}
    \item CRUD de receitas
    \item Categorizacao automatica
    \item Interface para gestao de receitas
    \item Validacoes e tratamento de erros
    \item Testes de integracao
\end{itemize}

\paragraph{Sprint 4 (Semanas 11-12): Modulo de Gastos Fixos}
\begin{itemize}
    \item CRUD de gastos fixos
    \item Sistema de categorias
    \item Lembretes de vencimento
    \item Interface de gestao
    \item Relatorios basicos
\end{itemize}

\paragraph{Sprint 5 (Semanas 13-14): Modulo de Custos Variaveis}
\begin{itemize}
    \item Sistema de orcamento por categoria
    \item Registro de gastos variaveis
    \item Alertas de limite
    \item Dashboard de acompanhamento
    \item Controle de saldo restante
\end{itemize}

\paragraph{Sprint 6 (Semanas 15-16): Calculo Principal e Dashboard}
\begin{itemize}
    \item Implementacao da logica de calculo principal
    \item Dashboard com visao geral
    \item Graficos e visualizacoes
    \item Sistema de poupanca e reservas
    \item Calculo de gastos livres
\end{itemize}

\subsubsection{Fase 3: Testes e Refinamentos (Semanas 17-20)}

\paragraph{Sprint 7 (Semanas 17-18): App Mobile}
\begin{itemize}
    \item Setup do projeto React Native
    \item Implementacao das telas principais
    \item Integracao com API backend
    \item Testes em dispositivos reais
    \item Push notifications basicas
\end{itemize}

\paragraph{Sprint 8 (Semanas 19-20): Testes e Polimento}
\begin{itemize}
    \item Testes end-to-end completos
    \item Correcao de bugs identificados
    \item Otimizacao de performance
    \item Testes de usabilidade
    \item Documentacao final
\end{itemize}

\subsubsection{Fase 4: Deploy e Lancamento (Semanas 21-22)}

\paragraph{Semana 21: Preparacao para Producao}
\begin{itemize}
    \item Configuracao do ambiente de producao
    \item Deploy da aplicacao web
    \item Build e teste do app mobile
    \item Configuracao de monitoramento
    \item Testes de carga
\end{itemize}

\paragraph{Semana 22: Lancamento}
\begin{itemize}
    \item Deploy final da versao 1.0
    \item Publicacao do app nas stores
    \item Documentacao do usuario
    \item Suporte inicial aos usuarios
    \item Coleta de feedback
\end{itemize}

\subsection{Dependencias e Riscos}

\subsubsection{Dependencias Criticas}
\begin{itemize}
    \item Aprovacao do design pelos stakeholders
    \item Disponibilidade da equipe de desenvolvimento
    \item Acesso aos servicos de cloud computing
    \item Aprovacao dos apps nas stores (Apple/Google)
\end{itemize}

\subsubsection{Buffer Time}
\begin{itemize}
    \item 20\% de buffer incluido em cada sprint
    \item Semana extra reservada para imprevistos
    \item Flexibilidade para ajustar escopo se necessario
\end{itemize}
