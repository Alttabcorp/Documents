\subsection{Modelo de Trabalho}

\subsubsection{Metodologia Ágil}
\begin{itemize}
    \item \textbf{Framework}: Scrum com adaptacoes
    \item \textbf{Sprints}: 2 semanas
    \item \textbf{Cerimonias}:
    \begin{itemize}
        \item Daily Standups (15 min)
        \item Sprint Planning (2h)
        \item Sprint Review (1h)
        \item Sprint Retrospective (1h)
    \end{itemize}
    \item \textbf{Ferramentas}: Jira, Confluence, Slack
\end{itemize}

\subsubsection{Trabalho Remoto/Hibrido}
\begin{itemize}
    \item Flexibilidade para trabalho remoto
    \item Reunioes presenciais semanais (se possivel)
    \item Comunicacao assincrona como padrao
    \item Documentacao centralizada e atualizada
    \item Horarios de core hours para colaboracao
\end{itemize}

\subsubsection{Contratacao}
\begin{itemize}
    \item \textbf{Internos}: Product Owner, Gerente de Projeto
    \item \textbf{Consultoria}: Tech Lead, Designers
    \item \textbf{Freelancers}: Desenvolvedores, QA
    \item \textbf{Terceirizacao}: DevOps, Security Specialist
\end{itemize}
